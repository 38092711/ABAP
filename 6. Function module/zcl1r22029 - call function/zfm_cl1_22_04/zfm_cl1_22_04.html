FUNCTION zfm_cl1_22_04.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_STLAN) TYPE  STLAN
*"     REFERENCE(IV_LANGU) TYPE  SY-LANGU
*"  EXPORTING
*"     REFERENCE(EV_ANTXT) TYPE  ANTXT
*"     REFERENCE(EV_MESSAGE) TYPE  CHAR50
*"     REFERENCE(EV_SUBRC) TYPE  SY-SUBRC
*"  EXCEPTIONS
*"      NOT_FOUND
*"----------------------------------------------------------------------

*       Global data declarations

  SELECT SINGLE antxt
    INTO ev_antxt
    FROM t416t
    WHERE stlan = iv_stlan
    AND spras = iv_langu.

  ev_subrc = sy-subrc.

  IF ev_subrc NE 0.
    ev_message = TEXT-e01.
    RAISE not_found.
  ENDIF.



ENDFUNCTION.
