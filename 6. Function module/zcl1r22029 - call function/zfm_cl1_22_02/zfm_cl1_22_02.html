FUNCTION zfm_cl1_22_02.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_HKONT) TYPE  HKONT
*"     REFERENCE(IV_KTOPL) TYPE  KTOPL
*"     REFERENCE(IV_LANGU) TYPE  SY-LANGU
*"  EXPORTING
*"     REFERENCE(EV_TXT50) TYPE  TXT50
*"     REFERENCE(EV_MESSAGE) TYPE  CHAR50
*"     REFERENCE(EV_SUBRC) TYPE  SY-SUBRC
*"  EXCEPTIONS
*"      NOT_FOUND
*"----------------------------------------------------------------------
*--select data

*       Global data declarations
  SELECT SINGLE txt50
    INTO  ev_txt50
    FROM skat
    WHERE saknr = iv_hkont
    AND ktopl EQ iv_ktopl
    AND spras = sy-langu.

*--set result data
  ev_subrc = sy-subrc.

  IF ev_subrc NE 0.
    ev_message = TEXT-e01.
    RAISE not_found.
  ELSE.
    ev_message = TEXT-e02.
  ENDIF.


ENDFUNCTION.
