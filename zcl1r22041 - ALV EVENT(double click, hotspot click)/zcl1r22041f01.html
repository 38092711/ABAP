*&---------------------------------------------------------------------*
*& Form get_documnet_header
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM get_documnet_header .

  CLEAR: gt_header.
  SELECT bukrs belnr gjahr budat bktxt waers
    INTO CORRESPONDING FIELDS OF TABLE gt_header
    FROM bkpf
    WHERE bukrs EQ pa_bukrs
    AND gjahr EQ pa_gjahr.

  IF gt_header IS INITIAL.
    MESSAGE s037 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

ENDFORM.

*& <--  p2        text
*&---------------------------------------------------------------------*
FORM display_screen .

  DATA ls_variant TYPE disvariant.

  IF go_container IS NOT BOUND.

    CLEAR: gt_fcat, gs_fcat.
    PERFORM set_field_catalog USING: 'X' 'BUKRS' 'BKPF' 'C' ' ',
                                     'X' 'BELNR' 'BKPF' 'C' ' ', "전표번호
                                     'X' 'GJAHR' 'BKPF' 'C' ' ',
                                     ' ' 'BUDAT' 'BKPF' 'C' ' ',
                                     ' ' 'BKTXT' 'BKPF' ' ' ' ',
                                     ' ' 'WAERS' 'BKPF' 'C' ' '.

    PERFORM set_layout.
    PERFORM create_object.

*--화면이 뜨기 전에 (first_display) ALV Events를 적용해야 함.
    SET HANDLER : lcl_event_handler=>double_click FOR go_alv_grid, "static method
                  go_event->hotspot_click         FOR go_alv_grid. "instance method


    ls_variant-report = sy-repid. "current report id
    ls_variant-handle = 'ALV1'. "alv id

    CALL METHOD go_alv_grid->set_table_for_first_display
      EXPORTING
        is_variant      = ls_variant
        i_save          = 'A'
        i_default       = 'X'
        is_layout       = gs_layout
      CHANGING
        it_outtab       = gt_header
        it_fieldcatalog = gt_fcat.

  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*

*&      --> P_
*&      --> P_
*&      --> P_
*&      --> P_
*&---------------------------------------------------------------------*
FORM set_field_catalog  USING pv_key pv_field pv_table pv_just pv_emph.

  gs_fcat-key = pv_key.
  gs_fcat-fieldname = pv_field.
  gs_fcat-ref_table = pv_table.
  gs_fcat-just = pv_just.
  gs_fcat-emphasize = pv_emph.

  CASE pv_field.
    WHEN 'BELNR'.
      gs_fcat-hotspot = abap_true.
  ENDCASE.

  APPEND gs_fcat TO gt_fcat.
  CLEAR gs_fcat.

ENDFORM.

FORM handle_double_click  USING pv_row pv_column.

*--event가 발생한 행의 정보를 읽는다.
*--click event인 경우 read table은 공식처럼 같이 가기
  CLEAR gs_header.
  READ TABLE gt_header INTO gs_header INDEX pv_row.

*--case문으로 선택된 컬럼마다 다른 기능을 하도록 처리할 수 있음
  CASE pv_column.
    WHEN 'BUKRS'.
    WHEN 'BELNR'.
  ENDCASE.


ENDFORM.

FORM handle_hotspot_click USING pv_row_id pv_column_id.

*--event가 발생한 행의 정보를 읽는다.
  CLEAR gs_header.
  READ TABLE gt_header INTO gs_header INDEX pv_row_id.


**********************************************************************
* 1.사용자가 선택한 전표번호를 통해 전표조회 화면으로 이동
* 2.sap memory ID에 필요한 값을 세팅한다
**********************************************************************
  SET PARAMETER ID: 'BUK' FIELD gs_header-bukrs,
                    'BLN' FIELD gs_header-belnr,
                    'GJR' FIELD gs_header-gjahr.

  "필드에는 변수가 아니라 어떤 값이든 상관 없음
*  SET PARAMETER ID 'GJR' FIELD sy-datum(4).
*  SET PARAMETER ID 'GJR' FIELD '2025'.
  "와 같이 시스템 필드 혹은 하드코딩도 가능.

  CALL TRANSACTION 'FB03' AND SKIP FIRST SCREEN.

ENDFORM.
